Создать нового пользователя:


CREATE ROLE username LOGIN;

ALTER ROLE username PASSWORD 'password';

GRANT ALL ON DATABASE dbname TO username;


Создать базу данных:


CREATE DATABASE pprog WITH owner = pprog template = template0 encoding = 'UTF8' lc_collate = 'ru_RU.UTF-8' lc_ctype = 'ru_RU.UTF-8';


Удалить базу данных:


DROP DATABASE dbname;
