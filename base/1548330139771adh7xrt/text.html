<script>

function OpenWin(){

window.open(

"https://google.com/",

"window_name",

"width=700, height=300,toolbar=no,location=no,directories=no,menubar=no,status=no"

);

}

</script>


<a href="javascript: OpenWin();">Ссылка</a>


<!--

https://google.com/ - адрес файла/сайта, первоначально загружаемого в новое окно

window_name - дается новому окну, чтобы впоследствии ссылаться на него из скрипта


Параметры:

width=300 - ширина нового окна (размер в пикселах)

height=300 - высота нового окна (размер в пикселах)

left=100 - абсцисса левого верхнего угла нового окна (размер в пикселах)

top=100 - ордината левого верхнего угла нового окна (размер в пикселах)

toolbar=no - вывод панели инструменов (1 / 0 / yes / no)

location=no - вывод адресной строки (1 / 0 / yes / no)

directories=no - вывод панели ссылок (1 / 0 / yes / no)

menubar=no - вывод строки меню (1 / 0 / yes / no)

scrollbars=no - вывод полос прокрутки (1 / 0 / yes / no)

resizable=no - возможность изменения размеров окна (1 / 0 / yes / no)

status=no - вывод строки статуса (1 / 0 / yes / no)

fullscreen=no - вывод на полный экран (1 / 0 / yes / no)

-->
